-- BIẾN TRẠNG THÁI
local _G_AutoFarm = false
local _G_AirFarm = false

local _G_AutoQuest = false
local _G_AutoHaki = false
local player = game.Players.LocalPlayer

-- 5. HÀM CHECK NHIỆM VỤ THEO CẤP (Sea 1)
function GetQuestName()
    local lv = player.Data.Level.Value
    if lv >= 0 and lv < 10 then return "BanditQuest1", "Bandit", 1
    elseif lv >= 10 and lv < 15 then return "MonkeyQuest1", "Monkey", 1
    elseif lv >= 15 and lv < 30 then return "MonkeyQuest1", "Gorilla", 2
    elseif lv >= 30 and lv < 40 then return "PirateIslandQuest1", "Pirate", 1
    elseif lv >= 40 and lv < 60 then return "PirateIslandQuest1", "Brute", 2
    elseif lv >= 60 and lv < 75 then return "DesertQuest", "Desert Bandit", 1
    elseif lv >= 75 and lv < 90 then return "DesertQuest", "Desert Officer", 2
    -- Bạn có thể thêm các đảo tiếp theo tại đây
    else return "BanditQuest1", "Bandit", 1
    end
end

-- 6. VÒNG LẶP CHÍNH
spawn(function()
    while task.wait() do
        if _G_AutoFarm then
            pcall(function()
                -- Tự động nhận Quest
                if _G_AutoQuest then
                    local questUI = player.PlayerGui.Main:FindFirstChild("Quest")
                    if questUI and not questUI.Visible then
                        local qName, mName, qNum = GetQuestName()
                        game:GetService("ReplicatedStorage").Remotes.CommF_:InvokeServer("StartQuest", qName, qNum)
                    end
                end

                -- Tìm quái
                local Monster = workspace.Enemies:FindFirstChildOfClass("Model")
                if Monster and Monster:FindFirstChild("HumanoidRootPart") then
                    -- Bay tới quái
                    local targetPos = Monster.HumanoidRootPart.CFrame
                    if _G_AirFarm then
                        player.Character.HumanoidRootPart.CFrame = targetPos * CFrame.new(0, 15, 0)
                    else
                        player.Character.HumanoidRootPart.CFrame = targetPos * CFrame.new(0, 0, 3)
                    end

                    -- Đánh siêu nhanh
                    if _G_FastAttack then
                        game:GetService("ReplicatedStorage").Remotes.Validator:FireServer(math.huge)
                        local tool = player.Character:FindFirstChildOfClass("Tool")
                        if tool then tool:Activate() end
                    end
                end
            end)
        end
    end
end)

-- 7. TỰ ĐỘNG HAKI
spawn(function()
    while task.wait(1) do
        if _G_AutoHaki and not player.Character:FindFirstChild("HasBuso") then
            game:GetService("ReplicatedStorage").Remotes.CommF_:InvokeServer("Buso")
        end
    end
end)
